#include <stdio.h>

using namespace std;

int solution(vector<int> d, int budget) {
    int answer = 0;
    
    sort(d.begin(), d.end());
    
    for(int n : d)
    {   
        if(n > budget) break;
        budget -= n;
        
        answer++;
    }
    
    return answer;
}
