int CheckMat(vector<vector<string>> park, int InHeight, int InWidth, int InMat)

    for(int i = 0; i < InMat; i++)
    {
        for(int j = 0; j < InMat; j++)

            bCheck |= (InWidth + j >= park[0].size());

                return -1;
        }
    }
    
    return InMat;
}
