        answer++;

        answer++;

        answer++;

}

#include <string>
#include <vector>
#include <iostream>

using namespace std;

int solution(vector<vector<string>> board, int h, int w) {
    int answer = 0;
    
    int dh[4] = {0, 1, -1, 0};
    int dw[4] = {1, 0, 0, -1};
    
    for(int i = 0; i < 4; i++)
    {
        int x = h + dh[i];
        int y = w + dw[i];
        
        bool bCheck = true;
        bCheck &= (x >= 0);
        bCheck &= (x < board.size());
        bCheck &= (y >= 0);
        bCheck &= (y < board[0].size());
            
        if(bCheck)
            if((board[h][w] == board[x][y]))
                answer++;
    }
    
    return answer;
}
